A Tree is vaild binary search tree must meet the requirment below.

  • If it is NULL, it is true.
  • If its left child tree is not vaild binay search tree, it's false.
  • If its right child tree is not valid binary search tree, is's false.
  • If its maxnum value of left child tree is bigger or equal to its value, it's false.
  • If its minnum value of right child tree is smaller or equal to its value, it's false.
  • Otherwise, it is true.
{% highlight cpp linenos %} /** * Definition for binary tree * struct TreeNode { * int val; * TreeNode *left; * TreeNode *right; * TreeNode(int x) : val(x), left(NULL), right(NULL) {} * }; */ class Solution { public: bool isValidBST(TreeNode *root) { if ( !root ) { return true; } if ( !isValidBST(root->left) ) { return false; } if ( !isValidBST(root->right) ) { return false; } if ( root->left && maxV(root->left) >= root->val ) { return false; } if ( root->right && minV(root->right) <= root->val ) { return false; } return true; } int minV(TreeNode *root) { int t = root->val; if ( root->left ) { t = min(t,minV(root->left)); } if ( root->right ) { t = min(t,minV(root->right)); } return t; } int maxV(TreeNode *root) { int t = root->val; if ( root->left ) { t = max(t,maxV(root->left)); } if ( root->right ) { t = max(t,maxV(root->right)); } return t; } }; {% endhighlight %}
