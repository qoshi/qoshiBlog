easy probem count sorting

because the range of keys is limited, so count sorting is the best way. {% highlight cpp linenos %} class Solution { public: void sortColors(int A[], int n) { int color[3]; memset(color,0,sizeof(color)); for ( int i = 0; i < n; i++ ) { color[A[i]]++; } int k = 0; for ( int i = 0; i < 3; i++ ) { for ( int j = 0; j < color[i]; j++ ) { A[k++] = i; } } } }; {% endhighlight %}
