Binary search the rows and then binary search the columns. {% highlight cpp linenos %} class Solution { public: bool searchMatrix(vector > &matrix, int target) { int n = matrix.size(); if ( !n ) { return false; } int m = matrix[0].size(); if ( !m ) { return false; } int l = 0; int r = n-1; int mid = 0; while ( l <= r ) { mid = l+(r-l)/2; if ( matrix[mid][0] <= target && target <= matrix[mid][m-1] ) { break; } if ( matrix[mid][0] >= target ) { r = mid-1; } if ( matrix[mid][m-1] <= target ) { l = mid+1; } } if ( l > r ) { return false; } int t = mid; l = 0; r = m-1; while ( l <= r ) { mid = l + (r-l)/2; if ( matrix[t][mid] == target ) { return true; } if ( matrix[t][mid] < target ) { l = mid+1; } if ( matrix[t][mid] > target ) { r = mid-1; } } return false; } }; {% endhighlight %}
