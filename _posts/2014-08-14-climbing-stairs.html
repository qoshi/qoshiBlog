Easy problem like Fibonacci, const space complexity solution. {% highlight cpp linenos %} class Solution { public: int climbStairs(int n) { int dp[3]; dp[0] = 1; dp[1] = 2; for ( int i = 2; i < n; i++ ) { dp[i%3] = dp[(i-1)%3]+dp[(i-2)%3]; } return dp[(n-1)%3]; } }; {% endhighlight %}
