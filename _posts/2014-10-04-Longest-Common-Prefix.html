Brute force way

Iterate for every string, find the result. {% highlight cpp linenos %} class Solution { public: string longestCommonPrefix(vector &strs) { int n = strs.size(); if ( 0 == n ) { return ""; } string str = strs[0]; int l,j; for ( int i = 1; i < n; i++ ) { l = str.length(); j = 0; while ( j < l && str[j] == strs[i][j] ) { j++; } str = str.substr(0,j); } return str; } }; {% endhighlight %}
