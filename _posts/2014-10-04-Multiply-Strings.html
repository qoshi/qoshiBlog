Pay attention to the carryer! {% highlight cpp linenos %} class Solution { public: string multiply(string num1, string num2) { int l1 = num1.length(); int l2 = num2.length(); int l3 = l1+l2; int result[l1+l2]; memset(result,0,sizeof(result)); int carryer = 0; int t = 0; for ( int i = 0; i < l1; i++ ) { carryer = 0; for ( int j = 0; j < l2; j++ ) { t = (num1[l1-1-i]-'0')*(num2[l2-1-j]-'0')+carryer+result[l3-1-i-j]; result[l3-1-i-j] = t%10; carryer = t/10; } result[l3-1-i-l2] += carryer; } string re = ""; char ch; int i = 0; while ( i < l3 && result[i] == 0 ) { i++; } if ( i == l3 ) { return "0"; } for ( ; i < l3; i++ ) { ch = '0'+result[i]; re += ch; } return re; } }; {% endhighlight %}
