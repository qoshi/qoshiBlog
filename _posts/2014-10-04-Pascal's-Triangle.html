One thing is important. You could not use t[j]=t[j-1]+t[j] to get the the jth element of the array. It was changed by you in the iteration last time. {% highlight cpp linenos %} class Solution { public: vector > generate(int numRows) { vector< vector< int > > result; result.clear(); if ( 0 == numRows ) { return result; } vector t; t.clear(); t.push_back(1); result.push_back(t); int temp,now; for ( int i = 2; i <= numRows; i++ ) { temp = 1; for i( int j = 1; j < i-1; j++ ) { now = t[j]; t[j] += temp; temp = now; } t.push_back(1); result.push_back(t); } return result; } }; {% endhighlight %}
